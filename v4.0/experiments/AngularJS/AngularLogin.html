About the Experiment

This experiment extends previous one by adding signing in/out feature. Once signed in, the user is able to access his profile page. Whenever a user logouts' he is redirected to the home page. Here a common LoginService is created that is shared by all controller which store information about the current user.

Code snippet:


    $scope.login = function() {
        var username = $scope.username;
        var password = $scope.password;
        var res = LoginService.login(username, password);
        if (res == true) {
            $scope.currentUser = LoginService.getCurrentUser();
            $scope.color = "btn btn-success";
        } else {
            $scope.color = "btn btn-danger";
        }
    }
    app.factory("LoginService", function () {
        var login = function (username, password) {
            for(var index in users) {
                if(users[index].username == username && users[index].password == password) {
                    currentUser = users[index];
                    return true;
                }
            }
        return false;
        }
    }
